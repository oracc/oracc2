% Physical Page Size (pps) module for DA TeX
% ==========================================
%
% Physical page sizes are set automatically if \setmarginsbytypefield
% or \settypefieldbymargins is used. The entry point for actually
% changing the paper size is deliberately difficult to use in an
% ordinary text to encourage use of the higher level interface.
%
% The format defaults to using \ppsUSletter.
%
\def\ppsUSletter{\gdef\ppsSiz@{%
  \global\physicalpageheight=11truein 
  \global\physicalpagewidth=8.5truein
}}
\def\ppsUSlegal{\gdef\ppsSiz@{%
  \global\physicalpageheight=14truein
  \global\physicalpagewidth=8.5truein
}}
\def\ppsAfour{\gdef\ppsSiz@{% DIN and ISO
  \global\physicalpageheight=297truemm
  \global\physicalpagewidth=210truemm
}}
\ifAFOURpaper\ppsAfour\else\ppsUSletter\fi
\endinput
