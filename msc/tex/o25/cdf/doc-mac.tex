% Macros for producing the DA documentation. These are separated out 
% because they're not needed for regular work and will probably not be
% much used beyond DA central.
\escapechar64
\def\showeffect#1{\leavevmode\hbox to2em{#1\hfil}{\vbfont\string#1}\endgraf}
\def\showaccent#1#2#3{\leavevmode
  \hbox to1in{\vbfont\string#1\hfil}%
  \hbox to1in{\vbfont\string#1\{#2\}\hfil}%
  \hbox to1in{#1 #2\hfil}%
  #3\endgraf}
\def\showfont#1#2#3{\leavevmode
  \hbox to1in{\vbfont\carat\string#1\hfil}%
  \hbox to1in{\vbfont\string#2\hfil}%
  #3\endgraf}
\def\escape#1{{\tt @#1}}
\def\fld#1{\%#1}
\def\showfield#1#2{\leavevmode\hbox to2em{\vbfont#1\hfil}%
  \hangindent\leftskip\advance\hangindent2em#2\endgraf}

\def\DATeX{DA\TeX}

\endinput
